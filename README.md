    CMakeLists.txt       -- 'Toplevel' CMake file, provided by catkin

      package.xml        -- Package manifest for package_1
    ...
    package_n/
      CMakeLists.txt     -- CMakeLists.txt file for package_n
      package.xml        -- Package manifest for package_n
